#include "HardwareSerial.h"

#define SERIAL_GET_WORDLENGTH(SERIAL_x)    ((uint16_t)(SERIAL_x&0xF000))
#define SERIAL_GET_PARITY(SERIAL_x)        ((uint16_t)(SERIAL_x&0x0F00))
#define SERIAL_GET_STOPBITS(SERIAL_x)      ((uint16_t)((SERIAL_x&0x00F0) << 8))
typedef struct
{
    TYPE_ENUM_LL_UART_BIT_WIDTH_SEL data_bit;
    TYPE_ENUM_LL_UART_PARITY_SEL parity_selection;
    TYPE_ENUM_LL_UART_STOP_BIT_SEL stop_bit;
} SERIAL_ConfigGrp_t;

static const SERIAL_ConfigGrp_t SERIAL_ConfigGrp[] =
{
    {LL_UART_WORD_LENGTH_8B, LL_UART_PARITY_EVEN, LL_UART_STOP_1B},   // SERIAL_7E1
    {LL_UART_WORD_LENGTH_8B, LL_UART_PARITY_EVEN, LL_UART_STOP_2B},   // SERIAL_7E2
    {LL_UART_WORD_LENGTH_8B, LL_UART_PARITY_ODD, LL_UART_STOP_1B},    // SERIAL_7O1
    {LL_UART_WORD_LENGTH_8B, LL_UART_PARITY_ODD, LL_UART_STOP_2B},    // SERIAL_7O2

    {LL_UART_WORD_LENGTH_8B, LL_UART_PARITY_NO, LL_UART_STOP_1B},   // SERIAL_8N1
    {LL_UART_WORD_LENGTH_8B, LL_UART_PARITY_NO, LL_UART_STOP_2B},   // SERIAL_8N2
    {LL_UART_WORD_LENGTH_9B, LL_UART_PARITY_EVEN, LL_UART_STOP_1B},   // SERIAL_8E1
    {LL_UART_WORD_LENGTH_9B, LL_UART_PARITY_EVEN, LL_UART_STOP_2B},   // SERIAL_8E2
    {LL_UART_WORD_LENGTH_9B, LL_UART_PARITY_ODD,  LL_UART_STOP_1B},   // SERIAL_8O1
    {LL_UART_WORD_LENGTH_9B, LL_UART_PARITY_ODD,  LL_UART_STOP_2B},   // SERIAL_8O2

    {LL_UART_WORD_LENGTH_9B, LL_UART_PARITY_NO, LL_UART_STOP_1B},   // SERIAL_9N1
    {LL_UART_WORD_LENGTH_9B, LL_UART_PARITY_NO, LL_UART_STOP_2B},   // SERIAL_9N2
};
/**
  * @brief  串口对象构造函数
  * @param  串口外设地址
  * @retval 无
  */
HardwareSerial::HardwareSerial(usart_type* usart)
    : _USARTx(usart)
    , _callbackFunction(NULL)
    , _rxBufferHead(0)
    , _rxBufferTail(0)
{
    memset(_rxBuffer, 0, sizeof(_rxBuffer));
}

/**
  * @brief  串口中断入口
  * @param  无
  * @retval 无
  */
void HardwareSerial::IRQHandler()
{

}

/**
  * @brief  串口初始化
  * @param  BaudRate: 波特率
  * @param  Config: 配置参数
  * @param  PreemptionPriority: 抢占优先级
  * @param  SubPriority: 从优先级
  * @retval 无
  */
void HardwareSerial::begin(
    uint32_t baudRate,
    SERIAL_Config_t config,
    uint8_t preemptionPriority,
    uint8_t subPriority
)
{
    gpio_type *GPIOx;
    TYPE_LL_GPIO_INIT gpio_init_struct = {0};
    uint16_t Tx_Pin, Rx_Pin;
    IRQn_Type USARTx_IRQn;

    if(_USARTx == UART0)
    {
        GPIOx = GPIOB;
        Tx_Pin = GPIO_Pin_6;
        Rx_Pin = GPIO_Pin_7;
        USARTx_IRQn = UART0_IRQn;
    }
    else if(_USARTx == UART1)
    {
        GPIOx = GPIOA;
        Tx_Pin = GPIO_Pin_2;
        Rx_Pin = GPIO_Pin_3;
        USARTx_IRQn = UART1_IRQn;
    }
    else
    {
        return;
    }

	gpio_init_struct.gpio_mode = LL_GPIO_MODE_AF;
	gpio_init_struct.gpio_type = LL_GPIO_TYPE_OUT_PP;
	gpio_init_struct.gpio_pupd = LL_GPIO_PUPD_NOPULL;
	gpio_init_struct.gpio_pin = Tx_Pin;

    ll_gpio_init(GPIOx, &gpio_init_struct);

	gpio_init_struct.gpio_pin = Rx_Pin;
	gpio_init_struct.gpio_pupd = LL_GPIO_PUPD_UP;
    ll_gpio_init(GPIOx, &gpio_init_struct);


    TYPE_LL_UART_INIT uart_init_data = {0};

    uart_init_data.baudrate = baudRate;
    uart_init_data.bit_width_sel = SERIAL_ConfigGrp[config].data_bit;
    uart_init_data.parity = SERIAL_ConfigGrp[config].parity_selection;
    uart_init_data.stop_bit_sel = SERIAL_ConfigGrp[config].stop_bit;
    uart_init_data.work_mode = LL_UART_WORK_MODE_FULL_DUPLEX;
    uart_init_data.timeout_len = 32;

    ll_uart_init(_USARTx, &uart_init_data);

    NVIC_EnableIRQ(USARTx_IRQn);
    NVIC_SetPriority(USARTx_IRQn, 1);
    ll_uart_enable(_USARTx);
}

/**
  * @brief  关闭串口
  * @param  无
  * @retval 无
  */
void HardwareSerial::end(void)
{
	usart_interrupt_enable(_USARTx, USART_RDBF_INT, FALSE);
    usart_enable(_USARTx, FALSE);
}

/**
  * @brief  串口中断回调
  * @param  Function: 回调函数
  * @retval 无
  */
void HardwareSerial::attachInterrupt(CallbackFunction_t func)
{
    _callbackFunction = func;
}

/**
  * @brief  获取可从串行端口读取的字节数
  * @param  无
  * @retval 可读取的字节数
  */
int HardwareSerial::available(void)
{
    return ((unsigned int)(SERIAL_RX_BUFFER_SIZE + _rxBufferHead - _rxBufferTail)) % SERIAL_RX_BUFFER_SIZE;
}

/**
  * @brief  读取传入的串行数据(字符)
  * @param  无
  * @retval 可用的传入串行数据的第一个字节 (如果没有可用的数据, 则为-1)
  */
int HardwareSerial::read(void)
{
    // if the head isn't ahead of the tail, we don't have any characters
    if (_rxBufferHead == _rxBufferTail)
    {
        return -1;
    }
    else
    {
        uint8_t c = _rxBuffer[_rxBufferTail];
        _rxBufferTail = (uint16_t)(_rxBufferTail + 1) % SERIAL_RX_BUFFER_SIZE;
        return c;
    }
}

/**
  * @brief  返回传入串行数据的下一个字节(字符), 而不将其从内部串行缓冲区中删除
  * @param  无
  * @retval 可用的传入串行数据的第一个字节 (如果没有可用的数据, 则为-1)
  */
int HardwareSerial::peek(void)
{
    if (_rxBufferHead == _rxBufferTail)
    {
        return -1;
    }
    else
    {
        return _rxBuffer[_rxBufferTail];
    }
}

/**
  * @brief  清空串口缓存
  * @param  无
  * @retval 无
  */
void HardwareSerial::flush(void)
{
    _rxBufferHead = _rxBufferTail;
}

/**
  * @brief  串口写入一个字节
  * @param  写入的字节
  * @retval 字节
  */
size_t HardwareSerial::write(uint8_t n)
{
    while(!usart_flag_get(_USARTx, USART_TDBE_FLAG)) {};
    usart_data_transmit(_USARTx, n);
    return 1;
}
